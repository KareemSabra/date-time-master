Installation

# Using bun
bun add time-master

# Using npm
npm install time-master

# Using yarn
yarn add time-master

# Using pnpm
pnpm add time-master

Quick Start

import { one, two } from 'time-master';

// Basic usage
const result1 = one(); // 1
const result2 = two(); // 2

// TypeScript support
const timeMaster: TimeMaster = {
    one,
    two,
};
